}

% ----------------------------------------------------------------------
% Beamer overrides
% ----------------------------------------------------------------------

\setbeamertemplate{section page}
{%
  \begin{centering}%
    \vfill%
    \begin{beamercolorbox}[sep=12pt,center]{section title}%
      \usebeamerfont{section title}\insertsection%
    \end{beamercolorbox}%
    \vfill%
  \end{centering}%
}

\setbeamercolor{structure}{fg=CfdBlue}

\setbeamertemplate{footnote}{
  \parindent 0pt%
  \noindent%
  \raggedright%
  \hbox to 1em{\insertfootnotemark\hfil}\insertfootnotetext
}

\AtBeginSection{\frame{\sectionpage}}
